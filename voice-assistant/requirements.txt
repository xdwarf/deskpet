# =============================================================================
# DeskPet Voice Assistant — Python dependencies (Raspberry Pi)
# =============================================================================
# Install with: pip install -r requirements.txt
# Run inside a virtual environment:
#   python3 -m venv venv && source venv/bin/activate
#
# NOTE: STT (Vosk) and AI (Gemini) run on the Docker server — see vosk-server/
# The Pi only needs audio I/O, wake word, TTS, casting, and HTTP client.
# =============================================================================

# MQTT client — same protocol as the ESP32 uses
paho-mqtt>=1.6.1

# Wake word detection
# openwakeword is lightweight and runs on Pi 3B+
openwakeword>=0.6.0

# Audio input (microphone via USB webcam)
pyaudio>=0.2.13

# Text-to-speech — Danish output generated on the Pi before casting
# gTTS: simple, uses Google TTS, requires internet
gTTS>=2.5.1
# Alternative: pyttsx3 for offline TTS (lower quality for Danish)
# pyttsx3>=2.90

# Chromecast / Google Nest speaker control — casts TTS audio directly by IP
pychromecast>=14.0.0

# HTTP requests — POSTs audio to /transcribe and text to /ask on the neo-brain server
requests>=2.31.0

# Config file parsing
PyYAML>=6.0.1

# Logging convenience
colorlog>=6.8.2

# NumPy — required by openwakeword
numpy>=1.24.0
